using namespace std;

const int N = 55 * 55 * 2, M = 20 * N;

int head[N], ver[M], next_[M], flow[M], value[M];
int d[N], incf[N], pre[N], v[N], tot = 0;
int n, k, S, T, maxflow, ans;

void add(int x, int y, int w, int c){
	ver[tot] = y, flow[tot] = w, value[tot] = c, next_[tot] = head[x], head[x] = tot ++;
	ver[tot] = x, flow[tot] = 0, value[tot] = -c, next_[tot] = head[y], head[y] = tot ++;
}

int num(int x, int y){
	return (x - 1) * n + y;
}

bool SPFA(){
	queue<int>q;
	memset(d, 0xcf, sizeof d);
	memset(v, 0, sizeof v);
	q.push(S); d[S] = 0; v[S] = 1;
	incf[S] = 1 << 30;
	while(q.size()){
		int x = q.front();
		v[x] = 0;
		q.pop();
		for (int i = head[x]; ~i; i = next_[i]){
			if (!flow[i]) continue;
			int j = ver[i];
			if (d[j] < d[x] + value[i]){
				d[j] = d[x] + value[i];
				incf[j] = min(incf[x], flow[i]);
				pre[j] = i;
				if (!v[j]) v[j] = 1, q.push(j);
			}
		}
	}
	if (d[T] == 0xcfcfcfcf) return false;
	return true;
}

void update(){
	int x = T;
	while(x != S){
		int i = pre[x];
		flow[i] -= incf[T];
		flow[i ^ 1] += incf[T];
		x = ver[i ^ 1];
	}
	maxflow += incf[T];
	ans += d[T] * incf[T];
}

void EK(){
	while(SPFA()) update();
	return ;
}

}
